create database escobar;

CREATE TABLE Login (
    Nome CHAR(50),
    CPF INT(11) PRIMARY KEY,
    Ano_NASC INT(8),
    Cep INT(8),
    UNIQUE (Nome) 
);

CREATE TABLE Produtos_da_Loja (
    Nome CHAR(50),
    id INT(6) PRIMARY KEY,
    data_insercao INT(8)
);

CREATE TABLE Estoque (
    Produto_id INT(6),
    Quantidade INT(10),
    FOREIGN KEY (Produto_id) REFERENCES Produtos_da_Loja(id)
);

CREATE TABLE Forma_de_Pagamento (
    id INT(2) PRIMARY KEY,
    método CHAR(20)
);

CREATE TABLE Dados_da_Compra (
    Nome_Cliente CHAR(50),
    Horário INT(4),
    Form_de_Pag INT(10),
    Token INT(10),
    FOREIGN KEY (Form_de_Pag) REFERENCES Forma_de_Pagamento(id),
    FOREIGN KEY (Nome_Cliente) REFERENCES Login(Nome)
);
